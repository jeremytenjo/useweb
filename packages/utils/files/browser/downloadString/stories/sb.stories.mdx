# downloadString

Download strings from the browser.

export const Template = (args) => {
  function download() {
    downloadString(args.string, args.fileName)
  }
  const usage = `<button onClick={() => downloadString('${args.string}', '${args.fileName}')}>
  Download ${args.fileName}
<button/>`
  return (
    <FunctionUsageExample
      usage={usage}
      result={<Button onClick={download} text={`Download ${args.fileName}`} />}
    />
  )
}
