# Firebase Functions

### Usage

```js
import useFirebaseFunction from '@useweb/use-firebase/useFirebaseFunction'

useFirebaseFunction({
  name: 'helloWorld',
})
```

```js
return (
  <FirebaseProvider
    functionsOptions={{
      onMessage: (message) => {
        console.log({ message })
      },
    }}
  >
    <App />
  </FirebaseProvider>
)
```

### Setup

1. Add provider to your app.

```js
import { initializeApp } from 'firebase/app'
import { getMessaging } from 'firebase/functions'
import { FirebaseProvider } from '@useweb/use-firebase'

const firebaseApp = initializeApp(firebaseConfig)
const functions = getMessaging()

return (
  <FirebaseProvider firebaseApp={firebaseApp} functions={functions}>
    <App />
  </FirebaseProvider>
)
```
